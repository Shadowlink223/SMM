ACTOR ProtectorDrone
{
	Health 300
	Radius 30
	Height 64
	Mass 150
	Speed 35
	Scale 0.8
	Painchance 32
	MONSTER
	+FLOORCLIP
	+NOTDMATCH
	+NOICEDEATH
	+THRUGHOST
	+NOINFIGHTING
	+FASTMELEE
	+FASTER
	+DONTMORPH
	PainSound "prodrone/pain"
	DeathSound "prodrone/death"
	ActiveSound "prodrone/roam"
	AttackSound "prodrone/attack"
	Obituary "%o was clawed to death by a Protector Drone."
    DamageFactor "Mace", 1.1
    DamageFactor "DragonClaw", 1.5
	
	States
	{
		Spawn:
			PRDR A 1 A_Look
			Loop
		See:
			PRDR BBCCDDEE 5 A_Chase
			Loop
		Melee:
			PRDR FFFF 2 A_FaceTarget
			PRDR G 8 A_CustomMeleeAttack((22), "prodrone/attack")
			PRDR H 8
			Goto See
		Pain:
			PRDR O 3
			PRDR O 3 A_Pain
			Goto See
		Pain.MightyBoot:
			TNT1 A 0 A_ChangeFlag("NOPAIN",1)
			TNT1 A 0 A_PlaySoundEx("duke/kickhit","body",0,0)
			PRDR O 2 A_FaceTarget
			TNT1 A 0 A_Recoil(14)
			PRDR O 16 A_Pain
			Goto See
		Death.DukeDamage:
		Death.DukePistol:
			TNT1 A 0 A_Jump(218,"Death")
			TNT1 A 0 A_GiveToTarget("DukeKillTaunt",1)
		Death.MightyBoot:
			TNT1 A 0 A_PlaySoundEx("duke/kickhit","body",0,0)
			TNT1 A 0 A_Jump(64,"Death")
			TNT1 A 0 A_GiveToTarget("DukeBootTaunt",1)
			goto Death
		Death.DukeExplosive:
			TNT1 A 0 A_JumpIfHealthLower(-3,"DukeGibStart")
			goto Death.DukeDamage
		DukeGibStart:
			TNT1 A 0 A_Jump(96,"XDeath")
			TNT1 A 0 A_GiveToTarget("DukeGibTaunt",1)
			Goto XDeath
		Death:
			TNT1 A 0 A_GiveToTarget("KillCount",1)
			PRDR P 5 A_Scream
			PRDR Q 5 A_NoBlocking
			PRDR RSTUVW 5
			PRDR W -1
			stop
		XDeath:
			TNT1 A 0 A_UnsetSolid
			TNT1 A 0 A_NoBlocking
			TNT1 A 0 A_GiveToTarget("KillCount",1)
			TNT1 A 0 A_SpawnDebris("DukeGibs1")
			TNT1 A 0 A_SpawnDebris("DukeGibs2")
			TNT1 A 0 A_SpawnDebris("DukeGibs3")
			TNT1 A 0 A_SpawnDebris("DukeGibs4")
			TNT1 A 0 A_SpawnDebris("DukeGibs5")
			TNT1 A 0 A_SpawnDebris("DukeGibs6")
			TNT1 A 0 A_PlaySound("duke/enemygib")
			Stop
		Death.ZorchPistol:
		Death.Zorch:
			TNT1 A 0 A_GiveToTarget("KillCount",1)
			PRDR I 0 A_Stop
			TNT1 A 0 A_ChangeFlag("DONTGIB",1)
			TNT1 A 0 A_PlaySoundEx("chex/zorch","soundslot5",0,0)
			PRDR I 3 A_SpawnItem("ZorchEffectSmall")
			TNT1 A 0 A_GiveInventory("EnemyZorched",1)
			PRDR I 3 A_SetTranslucent(0.9,0)
			PRDR I 0 A_Scream
			PRDR I 3 A_SetTranslucent(0.8,0)
			PRDR I 0 A_NoBlocking
			PRDR I 3 A_SetTranslucent(0.7,0)
			PRDR I 3 A_SetTranslucent(0.6,0)
			PRDR I 3 A_SetTranslucent(0.5,0)
			PRDR I 3 A_SetTranslucent(0.4,0)
			PRDR I 3 A_SetTranslucent(0.3,0)
			PRDR I 3 A_SetTranslucent(0.2,0)
			PRDR I 3 A_SetTranslucent(0.1,0)
			TNT1 A 1 A_SetTranslucent(1,0)
			TNT1 A -1
			stop
		Death.DukeIce:
			TNT1 A 0 A_PlaySound("duke/enemyfreeze")
			TNT1 A 0 A_GiveInventory("FreezethrowerWeakness")
			TNT1 A 0 A_Jump(218,"Ice")
			TNT1 A 0 A_GiveToTarget("DukeKillTaunt",1)
		Ice:
		Death.HexenIce:
		Death.Ice:
			TNT1 A 0 A_GiveToTarget("KillCount",1)
			TNT1 A 0 A_GiveInventory("EnemyZorched",1)
			TNT1 A 0 A_NoBlocking
			//TNT1 A 0 A_SpawnItem("GuardFrozen")
			PRDR K 0 A_FreezeDeath
			PRDR K 70 A_UnsetSolid
			PRDR K 0 A_IceGuyDie
			PRDR K 1 A_IceGuyDie
			TNT1 A -1
			stop
		Raise:
			PRDR WVUTSRQP 5
			Goto See
		ZorchRaise:
			TNT1 A 1 A_UnsetShootable
			TNT1 A 1 A_SpawnItem("SpawnFire2")
			TNT1 A 0 A_PlaySound("archvile/resummon")
			TNT1 A 1 A_TakeInventory("EnemyZorched",1)
			TNT1 A 0 A_SetTranslucent(0.1,0)
			PRDR A 3 A_SetTranslucent(0.2,0)
			PRDR A 3 A_SetTranslucent(0.3,0)
			PRDR A 3 A_SetTranslucent(0.4,0)
			PRDR A 3 A_SetTranslucent(0.5,0)
			PRDR A 3 A_SetTranslucent(0.6,0)
			PRDR A 3 A_SetTranslucent(0.7,0)
			PRDR A 3 A_SetTranslucent(0.8,0)
			PRDR A 3 A_SetTranslucent(0.9,0)
			PRDR A 3 A_SetTranslucent(1,0)
			PRDR A 1 A_SetShootable
			Goto See
	}
}
