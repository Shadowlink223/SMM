ACTOR QEnforcer
{
	Health 10115
	Radius 16
	Height 56
	Speed 15
	Mass 0x7FFFFFFF
	PainChance 200
	Scale 0.5
	WoundHealth 10035
	MONSTER
	+FLOORCLIP
    +NOTDMATCH
    +THRUGHOST
	+MISSILEMORE
	+MISSILEEVENMORE
	SeeSound "qenforcer/sight"
	PainSound "qenforcer/pain"
	DeathSound "qenforcer/death"
	ActiveSound "qenforcer/active"
	Dropitem "DefaultPinataPop"
	DropItem "QuakeCellBackpack"
	Obituary "%o was blasted by a Enforcer."
    DamageFactor "Mace", 1.1
    DamageFactor "DragonClaw", 1.5
    PainChance "MightyBoot", 255
	PainChance "MarathonStun", 255
	States
	{
		Spawn:
			QEN1 A 0
			QEN1 A 1 A_JumpIfInventory("AlreadySpawned",1,"Spawn2")
			QEN1 A 1 ACS_NamedExecuteAlways("SMMSpawningScript",0,1,0)
			QEN1 A 0 A_GiveInventory("AlreadySpawned",1)
		Spawn2:
			TNT1 A 0
			TNT1 A 0 A_JumpIfInventory("NoPatrolToken",1,"Stand")
			TNT1 A 0 A_Jump(32,"Walk")
		Stand:
			QEN1 ABCDEFG 4 A_Look
			Loop
		Walk:
			TNT1 A 0
			QEN1 H 6 A_Stop
			QEN1 IJKLMNOP 6 A_Look
			QEN1 QRS 6 A_Wander
			QEN1 T 6 A_Stop
			QEN1 UVW 6 A_Look
			Loop
		See:
			QEN1 XYZ 4 A_Chase
			QEN2 ABCDE 4 A_Chase
			Loop
		Missile:
			QEN2 FGHIJ 4 A_FaceTarget
			QEN2 K 3 Bright A_CustomMissile("QEnforcerLaser",45,12)
			QEN2 L 10 A_FaceTarget
			QEN2 K 4 Bright A_CustomMissile("QEnforcerLaser",45,12)
			QEN2 LMNO 4
			Goto See
		Pain:
			TNT1 A 0 A_ChangeFlag("NOPAIN",1)
			TNT1 A 0 A_Jump(96,"PainB","PainC","PainD")
			QEN2 P 3 A_Pain
			QEN2 QRS 3
			TNT1 A 0 A_JumpIfInventory("NoPainToken",0,2)
			TNT1 A 0 A_ChangeFlag("NOPAIN",0)
			TNT1 A 0
			Goto See
		PainB:
			QEN2 T 3 A_Pain
			QEN2 UVWX 3
			TNT1 A 0 A_JumpIfInventory("NoPainToken",0,2)
			TNT1 A 0 A_ChangeFlag("NOPAIN",0)
			TNT1 A 0
			Goto See
		PainC:
			QEN2 Y 3 A_Pain
			QEN2 Z 3
			QEN3 ABCDEF 3
			TNT1 A 0 A_JumpIfInventory("NoPainToken",0,2)
			TNT1 A 0 A_ChangeFlag("NOPAIN",0)
			TNT1 A 0
			Goto See
		PainD:
			QEN3 G 3 A_Pain
			QEN3 HIJKLMNOPQRSTUVWXY 3
			TNT1 A 0 A_JumpIfInventory("NoPainToken",0,2)
			TNT1 A 0 A_ChangeFlag("NOPAIN",0)
			TNT1 A 0
			Goto See
		Wound:
			QEN3 Z 1 A_JumpIfHealthLower(10000,2)
			TNT1 A 0 A_Die
			TNT1 A 0 A_Die("QuakeGib")
			Goto Death
		Wound.ZorchPistol:
		Wound.Zorch:
		Wound.ZorchKamikaze:
		Wound.PhasingZorch:
		Wound.ZorchSpectral:
			QEN2 S 1 A_Die("Zorch")
			Goto Death
		Death:
			TNT1 A 0 A_GiveToTarget("KillCount",1)
			TNT1 A 0 A_Jump(128,"FDeath")
			TNT1 A 0 A_Scream
			TNT1 A 0 A_NoBlocking
			QEN3 Z 3
			QEN4 ABCDEFGHIJKLM 3
			QEN4 M -1
			Stop
		FDeath:
			TNT1 A 0 A_Scream
			TNT1 A 0 A_NoBlocking
			QEN4 NOPQRSTUVWX 3
			QEN4 X -1
			Stop
		Death.ZorchPistol:
		Death.Zorch:
		Death.ZorchKamikaze:
		Death.PhasingZorch:
		Death.ZorchSpectral:
			TNT1 A 0 A_GiveToTarget("KillCount",1)
			QEN2 S 0 A_Stop
			TNT1 A 0 A_ChangeFlag("DONTGIB",1)
			TNT1 A 0 A_PlaySoundEx("chex/zorch","soundslot5",0,0)
			QEN2 S 3 A_SpawnItem("ZorchEffectSmall")
			TNT1 A 0 A_GiveInventory("EnemyZorched",1)
			QEN2 S 3 A_SetTranslucent(0.9,0)
			QEN2 S 0 A_Scream
			QEN2 S 3 A_SetTranslucent(0.8,0)
			QEN2 S 0 A_NoBlocking
			QEN2 S 3 A_SetTranslucent(0.7,0)
			QEN2 S 3 A_SetTranslucent(0.6,0)
			QEN2 S 3 A_SetTranslucent(0.5,0)
			QEN2 S 3 A_SetTranslucent(0.4,0)
			QEN2 S 3 A_SetTranslucent(0.3,0)
			QEN2 S 3 A_SetTranslucent(0.2,0)
			QEN2 S 3 A_SetTranslucent(0.1,0)
			TNT1 A 1 A_SetTranslucent(1,0)
			TNT1 A -1
			stop
		Death.QuakeGib:
		XDeath:
			TNT1 A 1 A_GiveToTarget("KillCount",1)
			TNT1 A 0 DamageThing(20000)
			TNT1 A 0 A_PlaySound("quake/gib", CHAN_VOICE)
			TNT1 AAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAA 0 A_SpawnItemEx("Blood", 0, 0, random(8, 56), random(-4, 4), 0, random(0, 6))
			TNT1 A 0 A_SpawnItemEx("QuakeSmallGibDropper",0,0,random(20,40),0,0,0,0,1)
			Stop
	}
}
