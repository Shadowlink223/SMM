ACTOR Rottweiler
{
	Health 10060
	Radius 24
	Height 64
	Speed 12
	Mass 0x7FFFFFFF
	PainChance 255
	Scale 0.5
	WoundHealth 10035
	//MeleeRange 70
	MONSTER
	+FLOORCLIP
    +NOTDMATCH
    +THRUGHOST
	+MISSILEMORE
	SeeSound "rottweiler/sight"
	PainSound "rottweiler/pain"
	DeathSound "rottweiler/death"
	ActiveSound "rottweiler/active"
	Dropitem "DefaultPinataPop"
	Obituary "%o was smashed by a Shambler."
    DamageFactor "Mace", 1.1
    DamageFactor "DragonClaw", 1.5
	DamageFactor "Explosive", 0.5
	DamageFactor "DukeExplosive", 0.5
	DamageFactor "FireExplosive", 0.5
    PainChance "MightyBoot", 255
	PainChance "MarathonStun", 255
	States
	{
		Spawn:
			QRT1 A 0
			QRT1 A 1 A_JumpIfInventory("AlreadySpawned",1,"Spawn2")
			QRT1 A 1 ACS_NamedExecuteAlways("SMMSpawningScript",0,1,0)
			QRT1 A 0 A_GiveInventory("AlreadySpawned",1)
		Spawn2:
			TNT1 A 0
			TNT1 A 0 A_JumpIfInventory("NoPatrolToken",1,"Stand")
			TNT1 A 0 A_Jump(32,"Walk")
		Stand:
			QRT1 ABCDEFGHI 3 A_Look
			Loop
		Walk:
			QRT1 JKLM 5 A_Wander
			QRT1 NOPQ 5 A_Look
			Loop
		See:
			QRT1 RSTUVWXYZ 3 A_Chase
			QRT2 ABC 3 A_Chase
			Loop
		Melee:
			QRT2 DEFG 3
			QRT2 H 3 A_CustomMeleeAttack(random(1,24),"rottweiler/attack")
			QRT2 IJK 3
			Goto See
		/*Missile:
			TNT1 A 0
			TNT1 A 0 A_PlaySound("shambler/attack",CHAN_WEAPON)
			QSH3 NOPQR 4 A_FaceTarget
			TNT1 A 0 A_PlaySound("shambler/boom",CHAN_WEAPON)
			QSH3 STU 4 A_CustomBulletAttack(0,0,1,10,"QuakeBulletPuff",512,CBAF_NORANDOM)
			QSH3 VWXY 4
			Goto See*/
		Pain:
			TNT1 A 0 A_ChangeFlag("NOPAIN",1)
			TNT1 A 0 A_Jump(128,"PainB")
			TNT1 A 0 A_Pain
			QRT2 UVWXYZ 3
			TNT1 A 0 A_JumpIfInventory("NoPainToken",0,2)
			TNT1 A 0 A_ChangeFlag("NOPAIN",0)
			TNT1 A 0
			Goto See
		PainB:
			TNT1 A 0 A_Pain
			QRT3 ABCDEFGHIJKLMNOP 3
			TNT1 A 0 A_JumpIfInventory("NoPainToken",0,2)
			TNT1 A 0 A_ChangeFlag("NOPAIN",0)
			TNT1 A 0
			Goto See
		Wound:
			QRT3 Q 1 A_JumpIfHealthLower(10000,2)
			TNT1 A 0 A_Die
			TNT1 A 0 A_Die("QuakeGib")
			Goto Death
		Death:
			TNT1 A 0 A_GiveToTarget("KillCount",1)
			TNT1 A 0 A_Scream
			TNT1 A 0 A_Jump(128,"DeathB")
			QRT3 QRSTUVWXY 3
			TNT1 A 0 A_NoBlocking
			QRT3 Y -1
			Stop
		DeathB:
			TNT1 A 0 A_Scream
			TNT1 A 0 A_Jump(128,"DeathB")
			QRT3 Z 3
			QRT4 ABCDEFGH 3
			TNT1 A 0 A_NoBlocking
			QRT4 H -1
			Stop
		Death.QuakeGib:	
		XDeath:
			TNT1 A 1 A_GiveToTarget("KillCount",1)
			TNT1 A 0 DamageThing(20000)
			TNT1 A 0 A_PlaySound("quake/gib", CHAN_VOICE)
			TNT1 AAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAA 0 A_SpawnItemEx("Blood", 0, 0, random(8, 56), random(-4, 4), 0, random(0, 6))
			TNT1 AA 0 A_SpawnItemEx("QuakeSmallGibDropper",0,0,random(20,40),0,0,0,0,1)
			Stop
	}
}
