ACTOR QKnight
{
	Health 10115
	Radius 24
	Height 64
	Speed 12
	Mass 0x7FFFFFFF
	PainChance 255
	Scale 0.5
	WoundHealth 10040
	//MeleeRange 70
	MONSTER
	+FLOORCLIP
    +NOTDMATCH
    +THRUGHOST
	+MISSILEMORE
	SeeSound "qknight/sight"
	PainSound "qknight/pain"
	DeathSound "qknight/death"
	ActiveSound "qknight/active"
	Dropitem "DefaultPinataPop"
	Obituary "%o was slashed by a Knight."
    DamageFactor "Mace", 1.1
    DamageFactor "DragonClaw", 1.5
	DamageFactor "Explosive", 0.5
	DamageFactor "DukeExplosive", 0.5
	DamageFactor "FireExplosive", 0.5
    PainChance "MightyBoot", 255
	PainChance "MarathonStun", 255
	States
	{
		Spawn:
			QKV1 A 0
			QKV1 A 1 A_JumpIfInventory("AlreadySpawned",1,"Spawn2")
			QKN1 A 1 ACS_NamedExecuteAlways("SMMSpawningScript",0,1,0)
			QKN1 A 0 A_GiveInventory("AlreadySpawned",1)
		Spawn2:
			TNT1 A 0
			TNT1 A 0 A_Jump(32,"Walk")
		Stand:
			QKN1 ABCDEFGHI 3 A_Look
			Loop
		Walk:
			QRT1 JKLM 5 A_Wander
			QRT1 NOPQ 5 A_Look
			QKN1 RSTU 5 A_Wander
			QKN1 VW 5 A_Look
			Loop
		See:
			QKN1 XYZ 3 A_Chase
			QKN2 ABCDE 3 A_Chase
			Loop
		Melee:
			QKN2 G 3 A_PlaySound("qknight/attack1", 5)
			QKN2 GHIJKLMNO 3 A_CustomMeleeAttack(random(1,9))
			QKN2 P 3
			Goto See
		MeleeB:
			QKN2 QRSTUV 3
			QKN2 WXYZ 3 A_CustomMeleeAttack(random(1,9),"rottweiler/attack")
		Pain:
			TNT1 A 0 A_ChangeFlag("NOPAIN",1)
			TNT1 A 0 A_Jump(128,"PainB")
			TNT1 A 0 A_Pain
			QKN3 ABC 3
			TNT1 A 0 A_JumpIfInventory("NoPainToken",0,2)
			TNT1 A 0 A_ChangeFlag("NOPAIN",0)
			TNT1 A 0
			Goto See
		PainB:
			TNT1 A 0 A_Pain
			QKN3 DEFGHIJKLMN 3
			TNT1 A 0 A_JumpIfInventory("NoPainToken",0,2)
			TNT1 A 0 A_ChangeFlag("NOPAIN",0)
			TNT1 A 0
			Goto See
		Wound:
			QRT3 O 1 A_JumpIfHealthLower(10000,2)
			TNT1 A 0 A_Die
			TNT1 A 0 A_Die("QuakeGib")
			Goto Death
		Death:
			TNT1 A 0 A_GiveToTarget("KillCount",1)
			TNT1 A 0 A_Scream
			TNT1 A 0 A_Jump(128,"DeathB")
			QKN3 OPQRSTUVWX 3
			TNT1 A 0 A_NoBlocking
			QKN3 X -1
			Stop
		DeathB:
			TNT1 A 0 A_Scream
			TNT1 A 0 A_Jump(128,"DeathB")
			QKN3 YZ 3
			QKN4 ABCDEFGHI 3
			TNT1 A 0 A_NoBlocking
			QKN4 I -1
			Stop
		Death.QuakeGib:
		XDeath:
			TNT1 A 1 A_GiveToTarget("KillCount",1)
			TNT1 A 0 DamageThing(20000)
			TNT1 A 0 A_PlaySound("quake/gib", CHAN_VOICE)
			TNT1 AAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAA 0 A_SpawnItemEx("Blood", 0, 0, random(8, 56), random(-4, 4), 0, random(0, 6))
			TNT1 AA 0 A_SpawnItemEx("QuakeSmallGibDropper",0,0,random(20,40),0,0,0,0,1)
			Stop
	}
}
