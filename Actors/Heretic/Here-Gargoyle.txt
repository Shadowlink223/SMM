ACTOR HereticImp3 : HereticImp2 
{
	States 
	{	
		Crash:
			TNT1 A 0 A_UnsetSolid
			IMPX I 7 A_ImpExplode
			IMPX J 7 A_Scream
			IMPX K 7 A_NoBlocking
			IMPX L -1
			Stop
		XCrash:
			IMPX X 7
			IMPX Y 7 A_NoBlocking
			IMPX Z -1
			Stop
		Death.Ice:
			TNT1 A 0 A_Gravity
			TNT1 A 0 A_GiveInventory("EnemyZorched",1)
			TNT1 A 0 A_GiveToTarget("KillCount",1)
			TNT1 A 0 A_NoBlocking
			IMPX G 0 A_UnsetSolid
			IMPZ A 70 A_GenericFreezeDeath
			IMPX G 0 A_IceGuyDie
			IMPZ A 1 A_IceGuyDie
			TNT1 A -1
			stop
	}
}
